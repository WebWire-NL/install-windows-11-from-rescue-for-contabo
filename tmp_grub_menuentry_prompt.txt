# GRUB menuentry for Windows Installer (BIOS/MBR)
menuentry "Windows Installer (BIOS) - ntldr fallback" {
    insmod part_msdos
    insmod ntfs
    insmod ntldr
    set root=(hd0,msdos2)
    ntldr /bootmgr
}
